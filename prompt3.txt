from manim import *

class Main(Scene):
    def construct(self):
        A = [-3, -1, 0]
        B = [3, -1, 0]
        C = [2, 3, 0]
        
        triangle = Polygon(A, B, C, color=BLUE, fill_opacity=0.2)
        self.play(Create(triangle))
        self.wait(1)
        
        base = LabeledLine(start=A, end=B, label="b=6", color=YELLOW)
        height = LabeledLine(start=[2, -1, 0], end=C, label="h=4", color=GREEN)
        self.play(Create(base))
        self.wait(1)
        self.play(Create(height))
        self.wait(1)
        
        derivation = [
            MathTex(r'A=\frac{1}{2}bh', font_size=36),
            MathTex(r'A=\frac{1}{2}\cdot 6 \cdot 4=12', font_size=36, color=GREEN)
        ]
        for i, line in enumerate(derivation):
            if i == 0:
                line.move_to(np.array([4, 3, 0]))
            else:
                line.next_to(derivation[i-1], DOWN, buff=0.2)
        for line in derivation:
            self.play(Write(line))
            self.wait(1)
